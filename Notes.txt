interface IFlxFSM<T>
{
  owner:T;
  group:IFlxFSMGroup<T>
  function changeState(State:IFlxFSMState<T>)
  function shutdown()
}

interface IFlxFSMState<T>
{
  function enter(FSM:IFlxFSM<T>)
  function update(FSM:IFlxFSM<T>)
  function exit(FSM:IFlxFSM<T>)
}

interface IFlxFSMGroup<T>
{
  function addState(State:IFlxFSMState<T>)
  signal:FlxTypedSignal<String-Void>
  function update()
}
